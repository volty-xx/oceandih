--[[ 
    NATIVE LUAU TABLE CLIENT-SIDE ASSEMBLER
    Optimized for volty-xx Github structural layout formatting.
--]]

local Workspace = game:GetService("Workspace")
local Players = game:GetService("Players")

local function spawnClientModelFromLuau(rawUrl)
    local req = request or syn.request or (http and http.request)
    if not req then 
        warn("Execution Denied: Your script executor lacks HTTP request functions.")
        return 
    end

    print("Downloading structural blueprint securely from GitHub...")
    local response = req({Url = rawUrl, Method = "GET"})
    if not response.Success then 
        warn("Failed to reach GitHub data server. Status code: " .. tostring(response.StatusCode))
        return 
    end

    -- Convert the raw text table data into an executable function block
    local rawText = response.Body
    local modelLoaderFunction, compileError = loadstring("return " .. rawText)
    
    if not modelLoaderFunction then
        warn("Compiler Error parsing GitHub file: " .. tostring(compileError))
        return
    end

    -- Run the function to retrieve the native Luau data table array
    local success, assetBlueprint = pcall(modelLoaderFunction)
    if not success or type(assetBlueprint) ~= "table" then
        warn("Runtime Error: The downloaded text could not resolve into a standard Luau Table array.")
        return
    end

    -- Build the parent container for your vehicle model
    local mainModel = Instance.new("Model")
    mainModel.Name = "Client_Assembled_Vehicle"
    
    local primaryPart = nil

    print("Reconstructing 3D structure piece-by-piece...")
    for _, item in ipairs(assetBlueprint) do
        local obj
        
        -- Safely extract vector table variables from data keys
        local sX, sY, sZ = table.unpack(item.Size or {1, 1, 1})
        local pX, pY, pZ = table.unpack(item.Position or {0, 0, 0})

        -- Handle special mesh configurations 
        if item.ClassName == "MeshPart" or item.ClassName == "Part" then
            obj = Instance.new("Part")
            if item.MeshId and item.MeshId ~= "" then
                local mesh = Instance.new("SpecialMesh")
                mesh.MeshType = Enum.MeshType.FileMesh
                mesh.MeshId = item.MeshId
                mesh.TextureId = item.TextureId or item.TextureID or ""
                mesh.Parent = obj
            end
        else
            pcall(function() obj = Instance.new(item.ClassName) end)
        end

        if obj then
            obj.Name = item.Name or "Part"
            obj.Size = Vector3.new(sX or 1, sY or 1, sZ or 1)
            obj.Position = Vector3.new(pX or 0, pY or 0, pZ or 0)
            obj.CanCollide = false
            obj.Anchored = true
            obj.Parent = mainModel
            
            if not primaryPart then primaryPart = obj end
        end
    end

    -- Automatically constraint-weld the entire chassis body together
    if primaryPart then
        mainModel.PrimaryPart = primaryPart
        for _, part in ipairs(mainModel:GetChildren()) do
            if part:IsA("BasePart") and part ~= primaryPart then
                part.Anchored = false
                local weld = Instance.new("WeldConstraint")
                weld.Part0 = primaryPart
                weld.Part1 = part
                weld.Parent = part
            end
        end
        primaryPart.Anchored = false
    end

    -- Teleport the completely built structure directly to your player coordinates
    local lPlayer = Players.LocalPlayer
    if lPlayer and lPlayer.Character and lPlayer.Character:FindFirstChild("HumanoidRootPart") then
        local playerPos = lPlayer.Character.HumanoidRootPart.Position
        mainModel:MoveTo(playerPos + Vector3.new(0, 8, 15))
    end

    mainModel.Parent = Workspace
    print("Success! Your vehicle has successfully compiled and deployed into the game workspace.")
end

-- Directly loads your native repository code file
spawnClientModelFromLuau("https://raw.githubusercontent.com/volty-xx/oceandih/refs/heads/main/da.txt")
